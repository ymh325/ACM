### 深拷贝与浅拷贝

浅拷贝：简单的赋值拷贝操作

深拷贝：在堆区重新申请空间，进行拷贝操作



当用户没有手动写拷贝函数时，系统会为类生成一个默认拷贝函数，并且这个函数为浅拷贝

**浅拷贝可能会导致同一处内存被重复释放**

```c++
#include <bits/stdc++.h>
using namespace std;
class per{
public:
    int a{};
    int *b{};

    per(int x,int height){
        a=x;
        b=new int(height);
        cout<<"有参构造函数"<<endl;
    }
    per(const per &p){
        a=p.a;
       
        if (p.b!=nullptr)b=new int(*p.b);
        //进行深拷贝，为对象申请一个新的内存空间
        
        //如果是系统默认的拷贝构造函数，会直接
    	//b = p.b
        //例如 per p2 = p1
        //p2的b与p1的b会指向同一处内存空间
        
        cout<<"拷贝构造函数"<<endl;
    }
    ~per(){
        if (b!=nullptr){
            delete b;
            b=nullptr;
        }
        cout<<"析构函数"<<endl;
    }
};
void test(){
    per p1(1, 2);
    cout << p1.b << endl;
    per p2 = p1;
    cout << p2.b << endl;
}
int main(){
    test();
    return 0;
}
```
