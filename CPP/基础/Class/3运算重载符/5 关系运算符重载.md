### 关系运算符重载



**作用**：重载关系运算符，可以让两个自定义类型对象进行对比操作

```cpp
#include "iostream"
using namespace std;
class per{
public:
   int ma;
   per(){ma=0;}
   explicit per(int a){ma=a;}
   per(const per &p){ma=p.ma;}
    
   //两个重载函数的返回值一个都为bool 或则 int(返回1或0)
   //为了规范操作，可以声明 const函数，防止属性被修改
   bool operator==(const per &p) const{
       return p.ma!=ma;
   }
   bool operator!=(const per &p) const{
       return p.ma!=ma;
   }
};
ostream& operator<<(ostream &out,const per &p){
    out<<p.ma;
    return out;
}
int main() {
    per p1(1),p2(2);
    if (p1==p2)cout<<"p1与p2相等"<<endl;
    if (p1!=p2)cout<<"p1与p2不相等"<<endl;
    return 0;
}
```
