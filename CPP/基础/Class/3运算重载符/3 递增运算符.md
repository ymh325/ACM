### 递增运算符



作用：通过重载递增运算符，实现自己的整形数据



- **前置++**

```cpp
#include "iostream"
using namespace std;
class per{
public:
    int ma;
    per (){ma=1;}
    
    //注意，这里应该引用&，才能一直对一个sh
    per& operator++(){
        ma++;
        return *this;
    }

};
ostream& operator<<(ostream &out,per &p){
    out<<p.ma;
    return out;
}
int main() {
    per p;
    cout<<++(++p)<<endl;
    return 0;
}
```

- **后置++**

```cpp
#include "iostream"
using namespace std;
class per{
public:
    int ma;
    per (){ma=1;}
    per(const per &p){
        ma=p.ma;
        cout<<"copy"<<endl;
    }
    //注意，后置++的返回值不能是&，因为t在返回后已经被释放
    per operator++(int){
        per t=*this;
        ma++;
        return t;
    }
};								//（1）
ostream& operator<<(ostream &out,const per &p){
    out<<p.ma;
    return out;
}
int main() {
    per p;
    cout<<p++<<endl;
    return 0;
}
```

> 注意：
>
> 在（1）处，有两中写法：
>
> ```cpp
> const per &p
> //因为在C++规则中临时对象是右值，右值是不能被非 const 引用绑定的。
> //后置++函数返回的是一个临时对象，属于右值,所以需要加上const
> ```
>
> ```cpp
> per p
> ```
