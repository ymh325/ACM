### 全局函数做友元



```cpp
class home{
    
    //告诉编译器 func 全局函数是 home类的好朋友，可以访问类中的私有内容
    friend void func(home &p);
    
public:
    string setRoom;
private:
    string bedRoom;
public:
    home(){
        setRoom = "客厅";
        bedRoom = "卧室";
    }
    
};
void func(home &p){
    cout<<p.setRoom<<endl;
    cout<<p.bedRoom<<endl;
}
int main(){
    home p;
    func(p);
    return 0;
}
```
