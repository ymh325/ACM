# 同余最短路



**当出现形如：**

- 给定  `n` 个整数，求这  `n`  个整数能拼凑出多少的其他整数（n 个整数可以重复取）
- 给定  `n` 个整数，求这 `n` 个整数不能拼凑出的最小（最大）的整数
- 至少要拼几次才能才能拼出模 `k` 余 `p` 的数

以上这些我问题，可以使用同余最短路的方法。

```c++
#include <iostream>
#include <vector>
#include <queue>
using namespace std;
#define OFF ios::sync_with_stdio(false),cin.tie(0), cout.tie(0);
#define int long long
const int N = 1e5 +5; 
const int inf =  9.22e18 - 1; 
struct edge {int v,w;};
vector<edge> e[N];
priority_queue<pair<int ,int>> q;
int h,d[N],vis[N];
int x,y,z;
void dis(int mi){
    for(int i=0;i<=mi;i++)d[i] = inf; 
    d[0] = 0;
    q.emplace(0,0);
    while(q.size()){
        auto t = q.top();q.pop();
        int u = t.second;
        if(vis[u])continue;
        vis[u] =1;
        for(edge ed:e[u]){
            int v = ed.v, w = ed.w;
            if(d[v]>d[u]+w){
                d[v] = d[u] +w;
                q.emplace(-d[v],v);
            }
        }
    }
}
signed main(){
    OFF
    cin>>h;
    h--;
    cin>>x>>y>>z;
    int mi = min(x,min(y,z));
    int a = max(x,max(y,z)) ;
    int b = x+y+z-a-mi;
    for(int i=0;i<mi;i++){
        e[i].push_back({(i+a)%mi,a});
        e[i].push_back({(i+b)%mi,b});
    }
    dis(mi);
    int ans=0;
    for(int i=0;i<mi;i++){
        if(d[i]<=h)ans += (h-d[i])/mi + 1;
        // cout<<d[i]<<" ";
    }
    cout<<ans;
    return 0;
}
```
