//TODO 用树链剖分 和 线段树
#include <vector>
#include <iostream>
using namespace std;
const int N = 1e5+5;
#define OFF ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
#define lc u<<1
#define rc u<<1|1
vector<int> e[N];
int fa[N],son[N],sz[N],dep[N];
int w[N];
int wt[N],id[N],cnt;
int top[N];
void dfs1(int u,int fat){
    fa[u]=fat;dep[u]=dep[fat]+1;sz[u]=1;
    for(int v:e[u]){
        if(v == fat)continue;
        dfs1(v,u);
        sz[u]+=sz[v];
        if(sz[son[u]]<sz[v])son[u]=v;
    } 
}
void dfs2(int u,int t){
    top[u]=t;id[u]=++cnt;wt[cnt]=w[u];
    if(!son[u])return;
    dfs2(son[u],t);
    for(int v:e[u]){
        if(v == fa[u]||v == son[u])continue;
        dfs2(v,v);
    }
}
struct tree{
    int l,r;
    int sum,tag;   
}tr[N*4];
void pushup(int u){
    tr[u].sum = tr[lc].sum + tr[rc].sum;
}
void pushdown(int u){
    if(tr[u].tag){
        int c = tr[u].tag;
        tr[lc].sum += c*(tr[lc].r-tr[lc].l+1);
        tr[rc].sum += c*(tr[rc].r-tr[rc].l+1);
        tr[lc].tag += c;
        tr[rc].tag += c;
        tr[u].tag = 0;
    }
}
void build(int u,int l,int r){
    //* { l, r, sum, tag} 
    tr[u]={l,r,wt[r],0};
    if(l==r)return ;
    int mid = (l + r)>>1;
    build(lc,l,mid);
    build(rc,mid+1,r);
    pushup(u);
}
int query(int u,int l,int r){
    if(l<=tr[u].l && tr[u].r <= r){
        return tr[u].sum;
    }
    pushdown(u);
    int mid = (tr[u].l + tr[u].r)>>1;
    int res=0;
    if(l<=mid)res+=query(lc,l,r);
    if(r>mid)res+=query(rc,l,r);
    return res;
}
void updata(int u,int l,int r,int k){
    if(l<= tr[u].l && tr[u].r <= r){
        tr[u].sum+=k*(tr[u].r-tr[u].l+1);
        tr[u].tag+=k;
        return ;
    }
    pushdown(u);
    int mid = (tr[u].r+tr[u].l)>>1;
    if(l<=mid)updata(lc,l,r,k);
    if(r>mid)updata(rc,l,r,k);
    pushup(u);
}
int query_path(int u,int v){
    int res=0;
    while(top[u]!=top[v]){
        if(dep[top[u]]<dep[top[v]])swap(u,v);
        res+=query(1,id[top[u]],id[u]);
        u = fa[top[u]];
    }
    if(dep[u]<dep[v])swap(u,v);
    res += query(1,id[v],id[u]);
    return res;
}
int query_tree(int u){
    return query(1,id[u],id[u]+sz[u]-1);
}
void updata_path(int u,int v,int k){
    while(top[u]!=top[v]){
        if(dep[top[u]]<dep[top[v]])swap(u,v);
        updata(1,id[top[u]],id[u],k);
        u = fa[top[u]];
    }
    if(dep[u]<dep[v])swap(u,v);
    updata(1,id[v],id[u],k);
}
void updata_tree(int u,int k){
    updata(1,id[u],id[u]+sz[u]-1,k);
}
int n,m,r;
int main(){
    OFF
    cin>>n>>m>>r;
    for(int i=1;i<=n;i++){
        cin>>w[i];
    }
    for(int i=1;i<n;i++){
        int u,v ;cin>>u>>v;
        e[u].push_back(v);
        e[v].push_back(u);
    }
    dfs1(r,0);
    dfs2(r,r);
    build(1,1,n);
    while(m--){
        int op ,x,y,z;
        cin>>op;
        if(op==1){
            cin>>x>>y>>z;
            updata_path(x,y,z);
        }else if(op==2){
            cin>>x>>y;
            cout<<query_path(x,y)<<endl;
        }else if(op==3){
            cin>>x>>z;
            updata_tree(x,z);
        }else {
            cin>>x;
            cout<<query_tree(x)<<endl;
        }
    }
    return 0;
}
