## 可持久化线段树

> 另名 主席树



## 思想：

支持回退，访问之前版本的线段树。保存每次插入的历史版本



## 变量设定

```c++
#define N 1e5+5
#define lc(x) tr[x].ch[0]
#define rc(x) tr[x].ch[1]

int n,m,a[N];
vector<int> v;
struct node{
  int ch[2];
    int s;// 节点值域中有多少个数
}tr[N*22];
int root[N],idx;
// root[]将存放各个版本的根节点
```



## 初始化树 $O(n)$

$n$ 为值域大小

```C++
void build(int &x,int l,int r){
    x=++idx;
    if(l==r)return;
    int m=l+r>>1;
    build(lc(x),l,m);
    build(rc(x),m+1,r);
}
build(root[0],1,n);// n为值域最大值
```

## 建主席树 $O(nlogn)$

递归建立各个历史版本的线段树

**参数：**x为前一版本的节点指针，y为当前版的节点指针

y是**"传引用"**，通过子空间 y 值，给父空间的 `lc(y)`或 `rc(y)`赋值

```c++
void insert(int x,int &y,int l,int r,int v){
    y=++idx;tr[y]=tr[x];tr[y].s++;
    if(l==r)return;
    int m=l+r>>1;
    if(v<=m)insert(lc(x), lc(y), l, m, v);
    else insert(rc(x), rc(y), m+1, r, v);
}
for(int i=1;i<=n;i++){
    insert(root[i-1], root[i], 1, n, a[i]);
}
```



## 查询 $O(nlogn)$

查询：在主席树上查询区间 `[l,r]` 第 `k` 小

简化：求 `[1,r]` 区间内的第 `k` 小。找到插入 `r` 时的历史版本

在权值线段树上二分搜索

扩展：求 `[l,r]` 区间内的第 `k` 小。我们想到了 **前缀和**，

`[l,r]` 的信息= `[1，r]` 的信息 - `[1,l-1]`的信息。

参数：`x` 为前一版本的节点指针，`y` 为当前版本的节点指针。

```C++
int query(int x,int y,int l,int r,int k){
    if(l==r)return l;
    int m=(l+r)>>1;
    int s=tr[lc(y)].s-tr[lc(x)].s;
    if(k<=s)return query(lc(x),lc(y),l,m,k);
    else return query(rc(x),rc(y),m+1,r,k-s);
}

scanf("%d%d%d",&l,&r,&k);
query(root[i-1],root[i],1,n,k);
```







## 离散化

考虑离散化：值域很大 $(|a_i|≤10^9)$，而数量有限 $(1≤n≤2×10^5)$，问题需要用值域做下标。

离散化：把“稀疏的数值”映射成“紧密的下标”。也可以说，把“大值域”映射成“小值域”。

**离散化三部曲：排序、去重、二分找下标。**

```c++
for(int i=1;i<=n;i++){
    scanf("%d",%a[i]);
    v.push_back(a[i]);
}
// 离散化
sort(v.begin(),v.end());// O(nlogn)
v.erase(unique(v.begin(),v.end()),v.end());

//返回 x 的下标
int getid(int x){
    return lower_bound(v.begin(),v.end(),x)-v.begin()+1;
}

for(int i=1;i<=n;i++){
    insert(root[i-1],root[i],1,vn,getid(a[i]));
}

int id=query(root[l-1],root[r],1,vn,k)-1;
printf("%d\n",v[id]);
```





## 模板

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
// #define ll long long
#define lc(x) tr[x].ch[0]
#define rc(x) tr[x].ch[1]
const int N = 2e5 + 5;
vector<int> v;
int n,m,idx;
int a[N];
struct tree{
    int ch[2];
    int s;
}tr[N*22];
int root[N];
void build(int &x,int l,int r){
    x = ++idx;
    if(l==r)return;
    int mid = (l+r)>>1;
    build(lc(x),l,mid);
    build(rc(x),mid+1,r);
}
void insert(int x,int &y,int l,int r,int vel){
    y=++idx;tr[y]=tr[x];tr[y].s++;
    if(l==r)return;
    int mid=(l+r)>>1;
    if(vel<=mid)insert(lc(x), lc(y), l, mid, vel);
    else insert(rc(x), rc(y), mid+1, r, vel);
}
int query(int x,int y,int l,int r,int k){
    if(l==r)return l;
    int mid=(l+r)>>1;
    int s=tr[lc(y)].s-tr[lc(x)].s;
    if(k<=s)return query(lc(x),lc(y),l,mid,k);
    else return query(rc(x),rc(y),mid+1,r,k-s);
}
int getPos(int x){
    return  lower_bound(v.begin(),v.end(),x)-v.begin()+1;
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i = 1;i <= n;i++){
        scanf("%d", &a[i]);
        v.push_back(a[i]);
    }
    
    sort(v.begin(),v.end());
    v.erase(unique(v.begin(),v.end()),v.end());

    build(root[0],1,v.size());
    for(int i = 1;i <= n;i++){
        insert(root[i-1], root[i], 1,v.size(), getPos(a[i]));
    }
    while(m--){
        int l,r,k;
        scanf("%d%d%d",&l,&r,&k);
        int p = query(root[l-1], root[r], 1, v.size(), k);
        printf("%d\n", v[p-1]);
    }
    return 0;
}
```
