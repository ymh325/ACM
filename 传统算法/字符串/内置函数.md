- **`str.substr()`**

  ```cpp
  #include <iostream>
  #include <string>
  
  int main() {
      std::string binaryStr = "10110011";  // 原始二进制字符串
  
      // 场景1：截取前 4 个字符（从索引0开始，取4个）
      std::string sub1 = binaryStr.substr(0, 4);
      std::cout << "sub1: " << sub1 << std::endl;  // 输出：1011
  
      // 场景2：从索引3开始，截取到末尾（count用默认值npos）
      std::string sub2 = binaryStr.substr(3);
      std::cout << "sub2: " << sub2 << std::endl;  // 输出：10011
  
      // 场景3：从索引2开始，截取3个字符
      std::string sub3 = binaryStr.substr(2, 3);
      std::cout << "sub3: " << sub3 << std::endl;  // 输出：110
  
      // 场景4：count超出剩余字符（从索引6开始，剩余2个字符，却要取5个）
      std::string sub4 = binaryStr.substr(6, 5);
      std::cout << "sub4: " << sub4 << std::endl;  // 输出：11（自动截取到末尾）
  
      return 0;
  }
  ```

  



- **`str.replace()`

```cpp
#include <iostream>
#include <string>

int main() {
    std::string binaryStr = "10100011";  // 原始二进制串

    // 场景1：替换索引2开始的3个字符（"100"）为 "111"
    binaryStr.replace(2, 3, "111");
    std::cout << "场景1: " << binaryStr << std::endl;  // 输出：10111011（原"101 000 11"→"101 111 11"）

    // 场景2：替换索引0开始的2个字符（"10"）为 "01"
    binaryStr.replace(0, 2, "01");
    std::cout << "场景2: " << binaryStr << std::endl;  // 输出：01111011

    // 场景3：替换到字符串末尾（len设为字符串长度 - pos）
    binaryStr.replace(4, binaryStr.size() - 4, "0000");
    std::cout << "场景3: " << binaryStr << std::endl;  // 输出：01110000

    return 0;
}
```
