```cpp
#include <iostream>
#define ll long long
using namespace std;
int A[2005],B[2005], C[4005];
int main(){
    string s1,s2;
    cin>>s1>>s2;
    int la = s1.size(),lb = s2.size();
    for(int i=0;i<la;i++){
        A[i]=s1[la-i-1]-'0';
    }
    for(int i=0;i<lb;i++){
        B[i]=s2[lb-i-1]-'0';
    }
    for (int i = 0; i < la; ++i) {
        for (int j = 0; j < lb; ++j) {
            C[j+i]=C[j+i]+A[i]*B[j];
            C[j+i+1]+=C[j+i]/10; //* 这里不用对是否 >= 10 进行判断
            C[j+i]=C[j+i]%10; //* 如果小于10 ， x/10 == 0 , x%10 == x
        }
    }
    int len = la + lb;
    while(len && C[len-1] == 0)len--;
    for (int i = len-1; i >=0 ; --i) {
        cout<<C[i];
    }
    return 0;
}
```
